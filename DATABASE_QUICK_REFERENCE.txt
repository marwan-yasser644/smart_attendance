# PATTERN 1: Handle QR Code Attendance Submission
from flask import Blueprint, request, jsonify

bp = Blueprint('attendance', __name__)

@bp.route('/attend/<session_uuid>', methods=['POST'])
def submit_attendance(session_uuid):
    from app.models import AttendanceSession
    
    session = AttendanceSession.query.filter_by(session_uuid=session_uuid).first()
    if not session or not session.is_open:
        return {'error': 'Session closed'}, 400
    
    data = request.json
    result = AttendanceOperations.record_student_attendance(
        session_id=session.id,
        student_id=data['student_id'],
        student_name=data['student_name'],
        ip_address=request.remote_addr
    )
    return result, (200 if result['success'] else 409)


# PATTERN 2: Generate Report (Date Range)
@bp.route('/report/<course_code>')
def course_report(course_code):
    from datetime import datetime, timedelta
    
    end = request.args.get('end', datetime.utcnow().date().isoformat())
    days = request.args.get('days', 30, type=int)
    
    start = datetime.fromisoformat(end) - timedelta(days=days)
    
    report = AttendanceOperations.get_attendance_by_course_and_date(
        course_code=course_code,
        start_date=start,
        end_date=end
    )
    return jsonify(report)


# PATTERN 3: Student Attendance History
@bp.route('/student/<student_id>/attendance')
def student_attendance(student_id):
    days = request.args.get('days', 30, type=int)
    
    history = AttendanceOperations.get_student_attendance_history(
        student_id=student_id,
        days_back=days
    )
    
    return jsonify({
        'student_id': student_id,
        'records': history,
        'total': len(history)
    })
